Report Generation SQL Examples

Sample queries for generating key reports from the data.

Funnel by Channel

SELECT channel, COUNT(DISTINCT CASE WHEN status_code = 'S00' THEN request_id END) AS created, COUNT(DISTINCT CASE WHEN status_code = 'S08' THEN request_id END) AS viewed, COUNT(DISTINCT CASE WHEN status_code = 'S21' THEN request_id END) AS consent, COUNT(DISTINCT CASE WHEN status_code = 'S40' THEN request_id END) AS success FROM sharing_status_history GROUP BY channel;

SP Performance Scorecard

WITH request_outcomes AS ( SELECT request_id, sp_id, MAX(CASE WHEN status_code = 'S40' THEN 1 ELSE 0 END) AS is_success FROM sharing_status_history GROUP BY request_id, sp_id ) SELECT sp_id, COUNT(*) AS total_requests, SUM(is_success) AS successful, ROUND(AVG(is_success) * 100, 1) AS success_rate_pct FROM request_outcomes GROUP BY sp_id ORDER BY success_rate_pct DESC;

Step Latency Analysis

SELECT previous_status || ' → ' || status_code AS transition, COUNT(*) AS occurrences, ROUND(AVG(step_latency_ms)/1000, 2) AS avg_sec, ROUND(PERCENTILE_CONT(0.9) WITHIN GROUP (ORDER BY step_latency_ms)/1000, 2) AS p90_sec FROM sharing_status_history WHERE previous_status IS NOT NULL GROUP BY transition ORDER BY avg_sec DESC;
